    ```go
    import (
        "errors"
        "fmt"
        "io"
        "os"
        "path/filepath"

        v1 "github.com/google/go-containerregistry/pkg/v1"
    )

    // Store manages the on-disk cache of ext4 rootfs images. Path layout:
    //   {CacheDir}/{algorithm}/{hex}.ext4
    // The cache is content-addressed via the OCI manifest digest.
    type Store struct {
        cacheDir string
    }

    // NewStore constructs a Store rooted at the given cache directory.
    // The directory is NOT created here; call Init() to create it.
    func NewStore(cacheDir string) *Store { return &Store{cacheDir: cacheDir} }

    // Init creates the cache directory tree with 0o750 perms. Idempotent.
    func (s *Store) Init() error {
        if err := os.MkdirAll(s.cacheDir, 0o750); err != nil {
            return &CacheError{Op: "init", Cause: err}
        }
        return nil
    }

    // PathFor returns the cache path for a given manifest digest.
    func (s *Store) PathFor(h v1.Hash) string {
        return filepath.Join(s.cacheDir, h.Algorithm, h.Hex+".ext4")
    }

    // Exists reports whether a cached ext4 file exists for this digest.
    func (s *Store) Exists(h v1.Hash) bool {
        _, err := os.Stat(s.PathFor(h))
        return err == nil
    }

    // AtomicWrite writes src's contents to the cache path for h, using a
    // temp file + rename. The final file is 0o444 (read-only). Safe under
    // concurrent callers writing the same digest (last-rename-wins, content
    // identical by content-addressing).
    func (s *Store) AtomicWrite(h v1.Hash, src io.Reader) error { ... }
    ```
    AtomicWrite implementation:
    1. dst := s.PathFor(h); dir := filepath.Dir(dst)
    2. os.MkdirAll(dir, 0o750)
    3. tmp, _ := os.CreateTemp(dir, filepath.Base(dst)+".tmp-*")
    4. defer os.Remove(tmp.Name()) (no-op if rename succeeds)
    5. io.Copy(tmp, src); tmp.Sync(); tmp.Chmod(0o444); tmp.Close()
    6. os.Rename(tmp.Name(), dst)
    7. Wrap any error in `&CacheError{Op: "write", Cause: err}` (or "rename" for the final step).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| disk cache → local filesystem | Config-controlled RootfsCacheDir path; bounded by host FS perms |
| user-supplied OCI ref → ParseReference | Untrusted string input (future Plan 02 feeds this to crane.Pull over network) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | Store cache directory | mitigate | Init() creates dir 0o750 (owner + group only); AtomicWrite sets 0o444 on final file so images cannot be mutated in place |
| T-02-02 | Tampering | Concurrent writes to same digest path | mitigate | AtomicWrite uses os.CreateTemp + os.Rename pattern; first-writer partial files never visible; last-writer-wins is safe because content is digest-addressed (same bytes by construction) |
| T-02-03 | DoS | MaxImageSize unbounded | mitigate | Validate() enforces MinMaxImageSize=32 MiB lower bound; default ceiling 2 GiB. Actual enforcement during conversion is Plan 02's tar2ext4.MaximumDiskSize call |
| T-02-04 | Input Validation | Malformed OCI ref | mitigate | ParseReference wraps name.ParseReference which rejects empty/malformed refs before any network call in Plan 02 |
| T-02-05 | Info Disclosure | Predictable temp file names | mitigate | os.CreateTemp with "*" suffix generates random names — no predictable path for a co-tenant to target |
| T-02-06 | Tampering | Cache directory symlink escape via RootfsCacheDir | accept | Operator supplies RootfsCacheDir; path is trusted input from trusted operator config. Not user-controlled at runtime |
</threat_model>
