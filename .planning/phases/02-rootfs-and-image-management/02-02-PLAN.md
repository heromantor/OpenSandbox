    ```go
        opts = append(opts, crane.WithContext(ctx))

        return img, nil
    ```

    ```go
    import (
        "context"
        "fmt"
        "io"
        "strings"

        "github.com/Microsoft/hcsshim/ext4/tar2ext4"
        "github.com/google/go-containerregistry/pkg/crane"
        v1 "github.com/google/go-containerregistry/pkg/v1"
    )

    // Provisioner builds ext4 rootfs images from OCI references and caches
    // them via the Store. Provisioner is safe for concurrent use.
    type Provisioner struct {
        cfg     ProvisionerConfig
        store   *Store
        fetcher ImageFetcher
    }

    // NewProvisioner constructs a Provisioner with the given configuration.
    // Uses the production craneFetcher for registry pulls.
    func NewProvisioner(cfg ProvisionerConfig) (*Provisioner, error) {
        return newProvisionerWithFetcher(cfg, NewCraneFetcher())
    }

    // newProvisionerWithFetcher is the test-visible constructor that
    // allows injecting a custom ImageFetcher.
    func newProvisionerWithFetcher(cfg ProvisionerConfig, fetcher ImageFetcher) (*Provisioner, error) {
        cfg = cfg.withDefaults()
        if err := cfg.Validate(); err != nil {
            return nil, err
        }
        store := NewStore(cfg.RootfsCacheDir)
        if err := store.Init(); err != nil {
            return nil, err
        }
        return &Provisioner{cfg: cfg, store: store, fetcher: fetcher}, nil
    }

    // Store returns the underlying cache store (for callers that need
    // to enumerate cache entries or build paths from digests).
    func (p *Provisioner) Store() *Store { return p.store }

    // Provision resolves ref to a manifest digest, builds the ext4 image
    // if not already cached, and returns the absolute path to the ext4
    // file. Safe for concurrent callers.
    func (p *Provisioner) Provision(ctx context.Context, ref string) (string, error) {
        if strings.TrimSpace(ref) == "" {
            return "", &InvalidProvisionerConfigError{Field: "ref", Message: "must not be empty"}
        }
        parsed, err := ParseReference(ref)
        if err != nil {
            return "", err
        }

        platform, perr := parsePlatform(p.cfg.DefaultPlatform)
        if perr != nil {
            return "", perr
        }

        img, err := p.fetcher.Fetch(ctx, parsed.Canonical, platform)
        if err != nil {
            return "", err
        }
        digest, err := img.Digest()
        if err != nil {
            return "", fmt.Errorf("firecracker: image: digest: %w", err)
        }

        // Cache hit short-circuit.
        if p.store.Exists(digest) {
            return p.store.PathFor(digest), nil
        }

        // Miss: stream crane.Export -> tar2ext4 -> atomic write via Store.
        if err := p.buildAndStore(ctx, img, digest); err != nil {
            return "", err
        }
        return p.store.PathFor(digest), nil
    }

    func (p *Provisioner) buildAndStore(ctx context.Context, img v1.Image, digest v1.Hash) error {
        // Use an io.Pipe to stream tar -> ext4 through a scratch temp file,
        // then hand the scratch file contents to Store.AtomicWrite.
        //
        // We use a temp file (not in-memory) because tar2ext4 requires
        // io.ReadWriteSeeker and ext4 images can exceed memory budget.
        scratch, err := os.CreateTemp(p.cfg.RootfsCacheDir, digest.Hex+".ext4.scratch-*")
        if err != nil {
            return &CacheError{Op: "scratch", Cause: err}
        }
        scratchPath := scratch.Name()
        defer func() { _ = scratch.Close(); _ = os.Remove(scratchPath) }()

        pr, pw := io.Pipe()
        errCh := make(chan error, 1)
        go func() {
            // crane.Export flattens image layers into a single tar stream.
            exportErr := crane.Export(img, pw)
            _ = pw.CloseWithError(exportErr)
            errCh <- exportErr
        }()

        convErr := tar2ext4.ConvertTarToExt4(pr, scratch,
            tar2ext4.ConvertWhiteout(),
            tar2ext4.MaximumDiskSize(p.cfg.MaxImageSize),
        )
        if convErr != nil {
            return &Ext4ConvertError{Cause: convErr}
        }
        if exportErr := <-errCh; exportErr != nil {
            return &ImagePullError{Ref: digest.String(), Cause: exportErr}
        }
        if err := scratch.Sync(); err != nil {
            return &CacheError{Op: "sync", Cause: err}
        }
        if _, err := scratch.Seek(0, io.SeekStart); err != nil {
            return &CacheError{Op: "seek", Cause: err}
        }

        // Check ctx after the expensive operation.
        if err := ctx.Err(); err != nil {
            return err
        }

        return p.store.AtomicWrite(digest, scratch)
    }

    // parsePlatform parses "linux/amd64" into a *v1.Platform.
    func parsePlatform(s string) (*v1.Platform, error) {
        parts := strings.SplitN(s, "/", 2)
        if len(parts) != 2 || parts[0] == "" || parts[1] == "" {
            return nil, &InvalidProvisionerConfigError{
                Field:   "DefaultPlatform",
                Message: fmt.Sprintf("must be 'os/arch', got %q", s),
            }
        }
        return &v1.Platform{OS: parts[0], Architecture: parts[1]}, nil
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| OCI registry → crane.Pull | Remote, untrusted mutable tag; integrity anchor is the manifest digest |
| tar stream → tar2ext4 | Untrusted content (registry-supplied); ext4 image is the security perimeter |
| io.Pipe producer/consumer goroutines | Internal; error propagation via channel + CloseWithError |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-07 | Tampering | Registry MITM swaps image | mitigate | crane enforces HTTPS by default; manifest digest (SHA-256) is recorded as cache key so tag mutation between calls is detectable (different cache path) |
| T-02-08 | DoS | Malicious image with enormous total size | mitigate | tar2ext4.MaximumDiskSize(p.cfg.MaxImageSize) fails conversion before disk fills; default 2 GiB + configurable ceiling |
| T-02-09 | DoS | Malicious image with billion small files (inode exhaustion) | mitigate | tar2ext4 MaximumDiskSize bounds the ext4 metadata budget transitively; fails at metadata exhaustion before host-level impact |
| T-02-10 | Info Disclosure | OCI whiteout files (.wh.*) leak into final fs | mitigate | tar2ext4.ConvertWhiteout() option processes .wh. entries as deletions (verified from tar2ext4 source in RESEARCH.md) |
| T-02-11 | Tampering | Arbitrary platform/architecture spoofing | mitigate | parsePlatform enforces explicit OS/Architecture from ProvisionerConfig.DefaultPlatform; crane.WithPlatform passes the explicit platform — no host-arch autodetect path |
| T-02-12 | DoS | ctx cancellation leak during crane.Export | mitigate | crane.Export writes to io.Pipe; ctx cancel triggers pw.CloseWithError chain; goroutine exits on next write attempt. ctx.Err() re-checked after conversion before final AtomicWrite |
| T-02-13 | Supply Chain | Unsigned image with malicious binaries | accept | v1 scope: cosign/sigstore signature verification deferred (A9 in RESEARCH.md). Operator responsibility via trusted registry + private mirrors |
| T-02-14 | Info Disclosure | Scratch file exposes partial image | mitigate | os.CreateTemp with random suffix under cfg.RootfsCacheDir (0o750 dir from Plan 01); defer os.Remove guarantees cleanup; file mode is 0o600 by default |
</threat_model>
