```makefile
# Kernel build artifact management
#
# These targets download and verify the pinned guest kernel image used by
# Firecracker VMs. The kernel version, URL, and SHA256 checksum are pinned
# in kernel.go (DefaultKernelPatchVersion, DefaultKernelURL, DefaultKernelSHA256).
#
# The kernel is stored in artifacts/ which is gitignored. Re-running
# fetch-kernel is idempotent: if the file already exists and the checksum
# matches, no download occurs.

KERNEL_VERSION := 5.10.223
KERNEL_URL     := https://s3.amazonaws.com/spec.ccfc.min/firecracker-ci/v1.10/x86_64/vmlinux-$(KERNEL_VERSION)
KERNEL_DIR     := artifacts
KERNEL_FILE    := $(KERNEL_DIR)/vmlinux-$(KERNEL_VERSION)
KERNEL_SHA256  := $(shell awk '/DefaultKernelSHA256/ {gsub(/[",]/,""); print $$NF}' kernel.go)

fetch-kernel: $(KERNEL_FILE)
	@echo "Verifying SHA256 of $(KERNEL_FILE)..."
	@if [ -z "$(KERNEL_SHA256)" ]; then \
		echo "ERROR: DefaultKernelSHA256 is empty in kernel.go — cannot verify"; \
		exit 1; \
	fi
	@actual=$$(shasum -a 256 "$(KERNEL_FILE)" 2>/dev/null | awk '{print $$1}' || sha256sum "$(KERNEL_FILE)" | awk '{print $$1}'); \
	if [ "$$actual" != "$(KERNEL_SHA256)" ]; then \
		echo "ERROR: SHA256 mismatch for $(KERNEL_FILE)"; \
		echo "  expected: $(KERNEL_SHA256)"; \
		echo "  actual:   $$actual"; \
		echo "  Delete $(KERNEL_FILE) and re-run 'make fetch-kernel'."; \
		exit 1; \
	fi
	@echo "OK: kernel $(KERNEL_VERSION) verified at $(KERNEL_FILE)"

$(KERNEL_FILE):
	@mkdir -p $(KERNEL_DIR)
	@echo "Downloading $(KERNEL_URL)..."
	@curl -fsSL -o $@ $(KERNEL_URL) || (rm -f $@; echo "ERROR: download failed"; exit 1)
	@echo "Downloaded $@"

verify-kernel:
	@if [ ! -f "$(KERNEL_FILE)" ]; then \
		echo "ERROR: $(KERNEL_FILE) not found. Run 'make fetch-kernel' first."; \
		exit 1; \
	fi
	@$(MAKE) fetch-kernel
```

NOTE on the `KERNEL_SHA256 := $(shell awk ...)` line: this extracts the checksum value from `kernel.go` at Makefile-parse time. The awk script looks for the line containing `DefaultKernelSHA256`, strips quotes and commas, and prints the last whitespace-separated field (which is the hex string). This keeps the checksum as a single source of truth in kernel.go.

NOTE on `shasum -a 256` vs `sha256sum`: macOS ships `shasum` (BSD coreutils) while Linux ships `sha256sum` (GNU coreutils). The shell expression tries `shasum` first and falls back to `sha256sum`.

```go
// VerifyKernelChecksum computes the SHA256 of the file at path and compares
// it to expectedHex (hex-encoded, lowercase, no prefix). Returns nil on match.
// Returns an error describing the mismatch, path, or read failure otherwise.
//
// expectedHex of empty string is treated as an explicit error — the caller
// must pin a real checksum.
func VerifyKernelChecksum(path, expectedHex string) error {
    if expectedHex == "" {
        return fmt.Errorf("firecracker: kernel checksum: expected hex is empty")
    }
    f, err := os.Open(path)
    if err != nil {
        return fmt.Errorf("firecracker: kernel checksum: open %s: %w", path, err)
    }
    defer f.Close()

    h := sha256.New()
    if _, err := io.Copy(h, f); err != nil {
        return fmt.Errorf("firecracker: kernel checksum: read %s: %w", path, err)
    }
    actual := hex.EncodeToString(h.Sum(nil))
    if actual != expectedHex {
        return fmt.Errorf("firecracker: kernel checksum mismatch for %s: expected %s, got %s", path, expectedHex, actual)
    }
    return nil
}
```
